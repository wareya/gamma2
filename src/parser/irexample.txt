// types: ptr(...), funcptr(...), u8, i8, u16, i16, u32, i32, u64, i64, f32, f64
// composite types: structs, arrays

struct Vec2
{
    f32 x;
    f32 y;
}


f32 plus_one(f32 a)
{
    return a + 1.0f32;
}

funcptr(f32, (f32)) r_add_one()
{
    return plus_one;
}

f32 add(f32 a, f32 b)
{
    Vec2 vec;
    vec.x = a;
    vec.y = b;
    a = vec.x + vec.y - 1.0f32;
    return r_add_one()(a);
    //return r_add_one()(a + b);
}
